simulation_engine.cpp 
#include <random>
#include <cmath>

double simulate_terminal_price(
  double S0, double r, double sigma,
  double T, std::mt19937& gen
) {
  std::normal_distribution<double> Z(0.0, 1.0);
  return S0 * std::exp(
    (r - 0.5 * sigma * sigma) * T
    + sigma * std::sqrt(T) * Z(gen)
  );
}
